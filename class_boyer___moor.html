std::vector<int> Boyer_Moor::prefix_func(const std::string& s) {
std::vector<int> p(s.length());
int k = 0;
p[0] = 0;
for (int i = 1; i < s.length(); ++i) {
while (k > 0 && s[k] != s[i]) {
k = p[k - 1];
}
if (s[k] == s[i]) {
++k;
}
p[i] = k;
}
return p;
}
